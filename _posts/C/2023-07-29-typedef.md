# typedef function pointer
```cpp
#include <iostream>
int add(int a,int b)
{
    return a+b;
}
typedef int(*FP)(int,int);

int main()
{
    FP fp;
    fp=add;
    std::cout<<fp(1,2);
    
    return 0;
}
```
